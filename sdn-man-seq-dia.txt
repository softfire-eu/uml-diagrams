deactivate p1

SM --> EM : response
deactivate SM
EM -->[
deactivate EM

end

activate SM

activate p2

p1 --> SM : (API-URL, user-flow-tables)

p2 --> SM : (user-flow-tables, ??)
deactivate p2
SM --> EM
deactivate SM

EM -> U : (token, API-URL, user-flow-tables)
end
